?onOff a brick:On_Command ;

?zone a brick:Zone ; brick:hasPart  ?space ;

?zoneRef a ref:IFCReference ;
ref:ifcName ?ZoneID .

{{?zone brick:isRegulatedBy/brick:hasPoint | brick:isFedBy/brick:hasPoint ?onOff .}}

{{?space brick:isLocationOf/brick:hasPoint ?onOff .}} 

}}
